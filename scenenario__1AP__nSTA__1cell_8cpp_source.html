48 
49  // AP object
50  AP AP_obj;

55 
56  // STA object

58 
59  // STA coordinates
60  // init uniform real distibution between [0,1]
61  std::uniform_real_distribution<double> uni_distribution(-0.5, 0.5);
62  // set random device generator
63  std::random_device generator;
64  bool inside;
65  double coordi_STA[2] = {0,0};
66  for (int i=0;i<common_params.n_STA;i++)
67  {
68  while(!inside){
69  coordi_STA[0] = uni_distribution(generator)*common_params.dimensions[0];
70  coordi_STA[1] = uni_distribution(generator)*common_params.dimensions[1];
71  inside = is_inside_rectangle(common_params.dimensions, coordi_STA);
72  }
73  //cout<<"x: "<<coordi_STA[0]<<" y: "<<coordi_STA[1]<<endl;
74  // initialize the STA objects
75  STA_obj[i].STA_init(i,coordi_STA,common_params,(void *)&AP_obj);
76  // Associate STA to AP
77  AP_obj.ADD_STA((void *) &STA_obj[i]);
78  // ADD AP to STA listener group
79  STA_obj[i].add_listener((void*) &AP_obj,AP_obj.TYPE);
80  // ADD STA to AP listener group
81  AP_obj.add_listener((void*) &STA_obj[i],STA_obj[i].TYPE);
82  inside = false;
83  }
84 
85  // ADD all STAs to each STA listener group
86  for (int i=0;i<common_params.n_STA;i++){
87  for (int j=0;j<common_params.n_STA ;j++){
88  // ADD all STAs except itself
89  if(j!=i)
90  STA_obj[i].add_listener((void*) &STA_obj[j],STA_obj[j].TYPE);
91  }
92  }
93 
94  // initialize jump
95  int jump=0,jump1=0,jump2=0;;
96  //cout<<"init done \n";
97  // Start simulation
98 
99  while (total_tti<run_ttis && stop_signal_called==false){ // run simulation for run_ttis TTIs
100  // check buffers of AP
101  jump1=AP_obj.check_buffer();
102  if(jump1<0)
103  std::cout<<" jump AP is in error : "<<jump1<<std::endl;
104  jump=jump1;
105  // check buffers of STA
106  for (int i=0;i<common_params.n_STA;i++){
107  jump2=STA_obj[i].check_buffer();
108  if(jump2<0)
109  std::cout<<" jump STA in error: "<<jump2<<std::endl;
110  jump = jump>jump2?jump2:jump;
111  }
112 
113  // schedule and tx AP
114  AP_obj.sched_tx();
115 
116  // schedule and tx STA
117  for (int i=0;i<common_params.n_STA;i++){
118  STA_obj[i].sched_tx();}
119 
120  // check tx time AP
121  AP_obj.check_tx_time();
122  // check tx time STA
123  for (int i=0;i<common_params.n_STA;i++){
124  STA_obj[i].check_tx_time();}
125 
126  // Print the current time
127  std::cout<<'\r';
128  if(system_time%100000==0) // display time every 10ms
129  std::cout<<system_time*1e-6<<" s time elapsed TTIs: "<<total_tti;//<<std::endl;
130 
131  // increment system time
132  system_time++;
133 
134 
135  if(jump>0){
136  // jump at AP
137  AP_obj.jump_time(jump);
138  // jump at STA
139  for (int i=0;i<common_params.n_STA;i++){
140  STA_obj[i].jump_time(jump);}
141  // jump system time
142  system_time+=jump;
143  } // end of jump condition
144  } // end of while loop
145 
146  // results
147  std::cout<<std::endl;
148  // measured load
149  cout<<"________________________________Results_________________________________________"<<endl;
150  std::cout<<"Simulation Run time : "<<system_time*1e-6<<" seconds"<<std::endl;
151  int total_ul_files =0;
152  for (int i=0;i<common_params.n_STA;i++){
153  total_ul_files += STA_obj[i].total_files;}
154  std::cout<<"Measured Load : "<<((double)(AP_obj.total_files+total_ul_files))*common_params.file_size/((double)system_time)<<" Mbps"<<std::endl;
155  std::cout<<"Given load : "<<(double) common_params.Network_load_bps*1e-6<<" Mbps"<<std::endl;
156  double meas_dl_serv = (double) AP_obj.served_data/((double)system_time);
157  double meas_ul_serv = 0;
158  for (int i=0;i<common_params.n_STA;i++){
159  meas_ul_serv += (double) STA_obj[i].served_data/((double)system_time);}
160  std::cout<<"ToT served load : "<<meas_dl_serv+meas_ul_serv<<" Mbps"<<std::endl;
161  std::cout<<"DL served load : "<<meas_dl_serv<<" Mbps"<<std::endl;
162  std::cout<<"UL served load : "<<meas_ul_serv<<" Mbps"<<std::endl;
163  cout<<"________________________________END_________________________________________"<<endl;
