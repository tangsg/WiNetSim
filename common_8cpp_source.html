103 /*! finds if the packet is in collision given the \a SINR_dB in dB and \a MCS_index \n
104  * \param SINR double
105  * \param MCS_index integer
106  * \return \c true if collision or \c false if not collision
107  */
108 bool find_prb_collision(double SINR_db, int MCS_index) {
109  // init uniform real distibution between [0,1]
110  std::uniform_real_distribution<double> uni_distribution(0.0, 1.0);
111  // set random device generator
112  std::random_device generator;
113  // interpolate and find the BLER for the given SINR and MCS Index
114  double bler = Monotone_cubic_spline_interpolation(
115  &AWGN_11ac_8_MCS_SNR[MCS_index], &AWGN_11ac_8_MCS_BLER[MCS_index],
116  SINR_db);
117  // get a random number between [0,1]
118  double rnd_number = uni_distribution(generator);
119  // see if bler is less than the number generateed
120  if (bler < rnd_number) // no collision
121  return false;
122  else
123  // collision
124  return true;
125 }
126 /*! Monotone_cubic_spline_interpolation \n
127  * reference : https://en.wikipedia.org/wiki/Monotone_cubic_interpolation \n
128  * The function takes \a SNR_LUT values sorted in ascending order
129  * and \a BLER_LUT corresponding to \a SNR_LUT and checks the \a BLER_LUT values corresponding to \a SNR_dB \n
130  * if the value is in the range of \a SNR_LUT and not present in the data then the function does a monotone cubic spline interpolation. \n
131  * if the data is out of range then it gives the first or last value depending on the side the \a SNR_dB falls in.
132  * \param *SNR_LUT pointer to vector
133  * \param *BLER_LUT pointer to vector
134  * \param SNR_dB double
135  * \return BLER_est double
136  */
137 double Monotone_cubic_spline_interpolation(std::vector<double> *SNR_LUT,
138  std::vector<double> *BLER_LUT, double SNR_db) {
139  int i = 0;
140  // compute dx i.e SNR differences
141  std::vector<double> dx;
142  for (auto it = SNR_LUT->begin(); it != SNR_LUT->end() - 1; ++it) // runs from i = 1...n-1
143  {
144  dx.push_back(*(it + 1) - *(it));
145  }
146  // compute dy i.e BLER differences
147  std::vector<double> dy;
148  for (auto it = BLER_LUT->begin(); it != BLER_LUT->end() - 1; ++it) // runs from i = 1...n-1
149  {
150  dy.push_back(*(it + 1) - *(it));
151  }
152  // compute slopes dy/dx
153  std::vector<double> ms;
154  for (auto it = dx.begin(); it != dx.end(); ++it) // runs from i = 1...n-1
155  {
156  ms.push_back(dy[i] / (*it));
157  i++;
158  }
159  std::vector<double> c1s;
160  std::vector<double> c2s;
161  std::vector<double> c3s;
162 
163  // compute c1s first order coefficients
164  c1s.push_back(ms.front());
165  for (i = 0; i < ms.size() - 1; i++) {
166  if (ms[i] * ms[i + 1] < 0)
167  c1s.push_back(0);
168  else
169  c1s.push_back(
170  3 * (dx[i] + dx[i + 1])
171  / (((dx[i] + dx[i + 1] + dx[i + 1]) / ms[i])
172  + ((dx[i] + dx[i + 1] + dx[i]) / ms[i + 1])));
173  }
174  c1s.push_back(ms.back());
175 
176  // compute c2s and c3s 2nd and third order coeffs
177  for (i = 0; i < c1s.size() - 1; i++) {
178  c2s.push_back(
179  (ms[i] - c1s[i] - c1s[i] - c1s[i + 1] + ms[i] + ms[i]) / dx[i]);
180  c3s.push_back((c1s[i] + c1s[i + 1] - ms[i] - ms[i]) / (dx[i] * dx[i]));
181  }
182  double BLER_est = 0;
183  bool do_interpolation = true;
184 
185  // see if we can avoid interpolation
186  for (i = 0; i < SNR_LUT->size(); i++) {
187  if (*(SNR_LUT->data() + i) == SNR_db) {
188  BLER_est = *(BLER_LUT->data() + i);
189  do_interpolation = false;
190  break;
191  }
192  }
193 
194  if (do_interpolation) {
195  // see if the input is not in range
196  if (SNR_db < SNR_LUT->front()) {
197  BLER_est = BLER_LUT->front();
198  } else if (SNR_db > SNR_LUT->back()) {
199  BLER_est = BLER_LUT->back();
200  } else {
201  // start interpolation
202  for (i = 0; i < SNR_LUT->size(); i++) {
203  if (SNR_db < *(SNR_LUT->data() + i)) {
204  double y_low = *(BLER_LUT->data() + i - 1);
205  double x_low = *(SNR_LUT->data() + i - 1);
206  double diff = SNR_db - x_low;
207  BLER_est = y_low + (c1s[i - 1] * diff)
208  + (c2s[i - 1] * diff * diff)
209  + (c3s[i - 1] * diff * diff * diff);
210  break;
211  } // end of if
212  } // end of for
213  } // end of else
214  } // end of if
215  return BLER_est;
216 } // end of function
