174 int AP::check_buffer() {
175  int jump = 0, i = 0;
176  // exponential distribution for load
177  std::exponential_distribution<double> exp_distribution(load_peruser);
178  // Decrement time next file arrives
179 
180  for (auto it = Time_next_file_arrives.begin();
181  it != Time_next_file_arrives.end(); ++it) {
182  (*it) -= 1;
183  if (*it <= 0) { // Check if the file arrived
184  log('[',system_time,']'," AP: ",ID);
185  // push the time the file arrived
186  Time_file_arrived_DL[i].push_back(system_time);
187  // Find the next packet arrival time
188  Time_next_file_arrives[i] = (long long) std::round(
189  exp_distribution(generator));
190  // Push the arrived file into the buffer
191  Buffer_DL[i].push_back(file_size);
192  // Sum the file size to sum buff buffer size
193  SUM_BUFF_DL[i] += file_size;
194  // Increment the STATISTIC total files arrived
195  total_files += 1;
196  // Increment buffer indicator
197  BUF_Ind += 1;
198  // Sum the file size to total buffer size
199  TOT_BUFF += file_size;
200  log(" STA: ",i," buff_size ",Buffer_DL[i].size(),
201  " total files ",total_files," Buf Ind ",BUF_Ind," Tot buff ",
202  TOT_BUFF,'\n');
203  }
204  i++; // Next user
205  }
206  // update jump with the time next file arrives
207  jump = Time_next_file_arrives[0];
208  // find the minimum among all the users buffer
209  for (auto it = Time_next_file_arrives.begin();
210  it != Time_next_file_arrives.end(); ++it) {
211  jump = jump < *it ? jump : *it;
212  }
213 
214  if (contending == false && TX_ind == 0 && RX_ind == 0) {
215  // if not TX and not Contending and not RX
216  // then check the buffer and if there is data start contending
217  if (BUF_Ind > 0) {
218  // As there is data start contending
219  contending = true;
220  // set jump to zero as there is data to tx
221  jump = 0;
222  // set a uniform distribution to get values between [0 CW-1]
223  std::uniform_int_distribution<int> uni_distribution(0, CW - 1);
224  // get a random number from uniform distribution and multiply it with wifi_slot to get time in us
225  Counter = uni_distribution(generator) * wifi_slot;
226  // Set DIFS_Ind to zero as this is the start of DCF procedure
227  DIFS_Ind = false;
228  // Set the DIFS Counter to DIFS duration
229  DIFS_counter = DIFS_time;
230  // Initialize energy measurement to thermal noise
231  Energy_meas = Noise;
232  // Measure the energy from all interferers
233  if (!ACTIVE_Devices.empty()) { // compute only if there are interferers
234  // initialize a vector pointer to measure PL
235  std::vector<double> *PL = new std::vector<double>;
236  // Measure PL from all the interferer sources
237  PL_measure(ACTIVE_Devices, PL, fc, channel_type); // PL is in dB
238  for (auto it = PL->begin(); it != PL->end(); ++it) {
239  // Sum all the received power from all the sources where rx power is tx power*PL
240  Energy_meas = Energy_meas + Pt_W * std::pow(10, *it / 10);
241  } // end of for
242  } // end of if
243  log('[',system_time,']'," AP: ",ID," Energy_meas ",Energy_meas," Noise ",Noise,
244  " contending ",contending," TX_ind ",TX_ind," RX_ind ",RX_ind,
245  " Counter ",Counter," DIFS_Ind ",DIFS_Ind," DIFS_counter ",
246  DIFS_counter," Active ",ACTIVE_Devices.size(),'\n');
247  } // end of if
248  } // end of if
249  else if (contending == true) {
250  // if contending then do energy measurements
251  // set jump to 0
252  jump = 0;
253  // Initialize energy measurement to thermal noise
254  Energy_meas = Noise;
255  // Measure the energy from all interferers
256  if (!ACTIVE_Devices.empty()) { // compute only if there are interferers
257  // initialize a vector pointer to measure PL
258  std::vector<double> *PL = new std::vector<double>;
259  // Measure PL from all the interferer sources
260  PL_measure(ACTIVE_Devices, PL, fc, channel_type); // PL is in dB
261  for (auto it = PL->begin(); it != PL->end(); ++it) {
262  // Sum all the received power from all the sources where rx power is tx power*PL
263  Energy_meas = Energy_meas + Pt_W * std::pow(10, *it / 10);
264  } // end of for
265  delete PL;
266  } // end of if
267  } // end of if else
268  else {
269  // might be receiving might be transmitting
270  // set the jump to zero
271  jump = 0;
272  } // end of else
273  // return time jump value
274  return jump;
275 }

277 /* sched and transmit
278  * Based on the energy measuremnt from previous function
279  * Clear channel assessment is done
280  * based on CCA DCF procedures are done.
281  * If AP gets the channel then ot schedules a UE in round robin fashion and transmits
282  */
283 void AP::sched_tx() {
284  // DO DCF if contending
285  if (contending == true) {
286  // See if DIFS is done
287  if (DIFS_Ind == true) {
288  // check if the measured enrgy is less than WIFI sensing threshold
289  if (Energy_meas > WIFI_TH) {
290  // If energy meas id > sens thresh then bring the counter to multiple of wifi slots
291  Counter = Counter - (Counter % wifi_slot) + wifi_slot;
292  // Set DIFS Ind to 0
293  DIFS_Ind = false;
294  // Set the DIFS counter
295  DIFS_counter = DIFS_time;
296  } // end of if energy condition
297  else
298  // Decrement the counter by 1 us If energy meas is less than sensing threshold
299  Counter = Counter - 1;
300  // Check whether the counter is zero
301  if (Counter <= 0) {
302  // If counter is zero
303  // set contending to false
304  contending = false;
305  // Set DIFS indicator to zero
306  DIFS_Ind = false;
307  // Set DIFS counter
308  DIFS_counter = DIFS_time;
309  // Set TX indicater to 1
310  TX_ind = 1;
311  // Start Scheduling
312  // Set channel occupy time to zero;
313  Channel_occupy_time = 0;
314  // Do round Robin
315  if (CURRENT_STA_DL == Connected_UEs - 1) { // if current UE is last one pick the next one
316  CURRENT_STA_DL = 0; // pick the first UE
317  } else {
318  CURRENT_STA_DL += 1; // pick the next UE
319  } // end of else
320  // schedule the UE
321  while (Channel_occupy_time == 0) {
322  // check if the current UE has data
323  if (SUM_BUFF_DL[CURRENT_STA_DL] > 0) {
324  // increment the attempt
325  attempt[CURRENT_STA_DL] = attempt[CURRENT_STA_DL] + 1;
326  // check the time required
327  long long time =
328  std::ceil(
329  (double) SUM_BUFF_DL[CURRENT_STA_DL]
330  / WIFI_data_20us[MCS_DL[CURRENT_STA_DL]]);
331  // check whether the time required is less than the MAX tx opportunity
332  if (20 * time < MAX_TXOP) { // 20us is the minimum granularity
333  // update the channel occupy time
334  Channel_occupy_time = 20 * time;
335  // Update Allocated data
336  ALLOC_data_DL[CURRENT_STA_DL] =
337  SUM_BUFF_DL[CURRENT_STA_DL];
338  // update the pumped data
339  pumped_data += SUM_BUFF_DL[CURRENT_STA_DL];
340  } else {
341  // occupy the channel for max tx op
342  Channel_occupy_time = MAX_TXOP;
343  // update Allocated data
344  ALLOC_data_DL[CURRENT_STA_DL] =
345  WIFI_data_20us[MCS_DL[CURRENT_STA_DL]]
346  * MAX_TXOP / 20;
347  // update pumped data
348  pumped_data +=
349  WIFI_data_20us[MCS_DL[CURRENT_STA_DL]]
350  * MAX_TXOP / 20;
351  } // end of else
352  } else {
353  // Do round Robin
354  // if current UE is last one pick the next one
355  if (CURRENT_STA_DL == Connected_UEs - 1)
356  CURRENT_STA_DL = 0; // pick the first UE
357  else
358  CURRENT_STA_DL += 1; // pick the next UE
359  } // end of else
360  } // end of inf while
361  log('[',system_time,']'," AP: ",ID," MCS_DL "
362  ,MCS_DL[CURRENT_STA_DL], "CURRENT_STA_DL ",CURRENT_STA_DL, "occupy time ",
363  Channel_occupy_time,'\n');
364  // Notify the listening group about this transmission
365  notify_listener(TRANSMITTING);
366  } // end of counter zero
367  } else if (Energy_meas < WIFI_TH) { // if measured energy is less than the WIFI sensing threshold (1us)
368  // Decrement DIFS counter by 1us
369  DIFS_counter = DIFS_counter - 1;
370  // Check if the DIFS counter is zero
371  if (DIFS_counter <= 0) {
372  // Indicate that DIFS is done
373  DIFS_Ind = 1;
374  // initialize the DIFS counter (Redundant but to be sure)
375  DIFS_counter = DIFS_time;
376  } // end of if
377  } // end of enregy meas
378  } // End of if contending
379 } // End of function

399 /*Callback function for tx

510 void AP::callback_fintx(void* obj, std::string type) {
511 
512  thread_safe_mtx.lock(); //thread safe
513 
514  bool temp = 0;
515  if (type == "AP")
516  temp = 0;
517  if (type == "STA") {
518  // check if the STA is associated with AP
519  for (auto it = BS_UEs.begin(); it != BS_UEs.end(); ++it) {
520  if (obj == *it) {
521  temp = 1;
522  break;
523  }
524  }
525  }
526 
527  if (temp && RX_ind == 1 && CURRENT_STA_UL == ((STA*) obj)->ID) { // if STA is associated with AP and AP is receiving
528  // stop receiving
529  RX_ind = 0;
530  // Measure the PL
531  double PL_meas = 0;
532  if (!ACTIVE_Devices.empty()) { // compute only if there are interferers
533  // initialize a vector pointer to measure PL
534  std::vector<double> *PL = new std::vector<double>;

536  PL_measure(ACTIVE_Devices, PL, fc, channel_type);
537  for (auto it = PL->begin(); it != PL->end(); ++it) {
538  // Sum all the received power from all the sources where rx power is tx power*PL
539  PL_meas += Pt_W * std::pow(10, *it / 10);
540  }
541  delete PL;
542  } // end of active device
543  // compute interference + noise
544  Int_power += PL_meas * (system_time - local_time)
545  + Channel_occupy_time * Noise;
546  std::vector<double> d;
547  // calculate the distance between STA and serving AP
548  d.push_back(
549  std::sqrt(
550  std::pow(coordinates[0] - ((STA*) obj)->coordinates[0],
551  2)
552  + std::pow(
553  coordinates[1]
554  - ((STA*) obj)->coordinates[1],
555  2)

558  PL_measure(d, PL_sig, fc, channel_type);
559  // calculate the received signal power
560  Sig_power = Channel_occupy_time * Pt_W
561  * std::pow(10, PL_sig->front() / 10);
562  delete PL_sig;
563  // Calculate SINR and convert to dB
564  double SINR_db = 10 * std::log10(Sig_power / Int_power);
565  // Log the SINR for stats
566

572  bool collision = find_prb_collision(SINR_db, ((STA*) obj)->MCS_UL);
573  // update global variable total_tti marking finish of 1 tti
574  total_tti++;

576  // if packet not in collision
577  // update the success
