315 void populate_params_nAP_nSTA_ncell_multi_thread() {
316  // constants
317  // Temperature
318  double Temp = 270;
319  // boltzman constant
320  double k = 1.3806488 * 1e-23;
321  // Noise Figure
322  double NF = std::pow(10, 0.7);
323  // Bandwidth of a single subcarrier
324  double B_wifi = 20e6 / 64;
325  // Thermal Noise floor in Watts
326  double Noise = k * Temp * B_wifi * NF;
327  common_params.Noise = Noise;
328 
329  // Layout Dimensions
330  // Length of the room
331  double length = 100;
332  common_params.dimensions[0] = length;
333  // Breadth of the room
334  double breadth = 100;
335  common_params.dimensions[1] = breadth;
336  // Height of AP from ground
337  double height_AP = 0;
338  common_params.height_AP = height_AP;
339  // Height of STA from ground
340  double height_STA = 0;
341  common_params.height_STA = height_STA;
342  // System Specific
343  // Total cells in the layout
344  int NCell = 1;
345  common_params.NCell = NCell;
346  // Total number of APs percell
347  int n_AP = 2;
348  common_params.n_AP = n_AP;
349  // Total number of nodes per AP
350  int n_STA = 3;
351  common_params.n_STA = n_STA;
352  // N threads
353  int N_threads = (NCell*n_AP*n_STA)+(NCell*n_AP);
354  common_params.N_threads = N_threads;
355  // Bandwidth of the system
356  double BW = 20;
357  common_params.BW = BW;
